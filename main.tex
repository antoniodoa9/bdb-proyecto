%% Fix listings + XeLaTeX/fontspec: \setmonofont carga Iosevka vía fontspec pero
%% \ttfamily no renderiza en ningún contexto (listings ni inline). Se define una
%% familia explícita con \newfontfamily (que sí funciona) y se redirigen tanto
%% \lstset como \texttt para usarla.
\newfontfamily{\codefont}[Scale=0.88]{Iosevka.ttf}
\lstset{basicstyle=\codefont\small}
\renewcommand{\texttt}[1]{{\codefont\small #1}}

%% Mismo problema con las URL de la bibliografia: \url usa \ttfamily, que no
%% renderiza con este montaje de fontspec y dejaba las direcciones en blanco
%% (solo se veia el punto final). Se redirige \UrlFont a la familia que si
%% funciona para que las URL de las referencias vuelvan a aparecer.
\renewcommand{\UrlFont}{\codefont}

%% Colocacion de floats: se relajan las fracciones por defecto para que las

\renewcommand{\bottomfraction}{0.8}
\renewcommand{\textfraction}{0.07}
\renewcommand{\floatpagefraction}{0.7}

\setcounter{bottomnumber}{2}
\setcounter{totalnumber}{5}
