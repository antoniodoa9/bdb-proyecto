SELECT 
    pa.codigo_hash,
    pa.anio_nacimiento,
    pa.sexo_biologico,
    pa.grupo_sanguineo,
    pa.region_sanitaria,
    ec.fecha_evento,
    ec.tipo_evento,
    ec.codigo_cie10,
    ec.descripcion_cie10,
    ec.especialidad
FROM paciente_anonimo pa
INNER JOIN evento_clinico ec ON ec.id_paciente = pa.id_paciente
WHERE ec.codigo_cie10 LIKE 'J%' -- Filtramos por respiratorio para acotar el volumen
ORDER BY pa.codigo_hash, ec.fecha_evento ASC;


SELECT 
    codigo_hash_paciente,
    fecha_inicio_baja,
    dias_totales_baja,
    codigo_cie10_causa,
    tipo_contingencia
FROM historico_bajas_medicas
ORDER BY codigo_hash_paciente;


USE PROYECTO;

SELECT 
    pa.codigo_hash,
    pa.anio_nacimiento,
    pa.sexo_biologico,
    t.fecha_hora_llegada,
    t.nivel_triaje_manchester,
    t.motivo_consulta,
    t.saturacion_oxigeno,
    t.estado_actual
FROM paciente_anonimo pa
INNER JOIN triaje_urgencias_actual t ON pa.codigo_hash = t.codigo_hash_paciente
WHERE pa.anio_nacimiento <= 1966 
  AND t.nivel_triaje_manchester <= 2
ORDER BY t.nivel_triaje_manchester ASC, t.fecha_hora_llegada ASC;




xquery version "3.1";

let $urgencias := doc("/db/PROYECTO/urgencias_criticas_semanticas.xml")
let $historiales := doc("/db/PROYECTO/historiales_semanticos_maximo.xml")
let $bajas := doc("/db/PROYECTO/bajas_semanticas.xml")

return
<Informe_Pacientes>
{
    (:Iteramos sobre cada paciente que está ahora mismo en urgencias :)
    for $paciente_urg in $urgencias//Paciente_Critico
    
    (:Extraemos su ID y sus datos :)
    let $id := $paciente_urg/@identificador
    let $nivel := xs:integer($paciente_urg/Estado_Clinico_Actual/@nivel_prioridad)
    let $motivo := $paciente_urg/Estado_Clinico_Actual/Motivo_Principal/text()
    
    (: Buscamos cuántos eventos tiene ese mismo ID en el historial :)
    let $eventos := $historiales//Paciente[@id = $id]/Fichero_Eventos/Evento_Medico
    let $num_eventos := count($eventos)
    
    (: Buscamos a ese mismo ID en las bajas y sumamos los días :)
    let $bajas_paciente := $bajas//Expediente_Laboral[@id_paciente = $id]/Incapacidad_Temporal
    let $total_dias_baja := sum($bajas_paciente/Duracion_Dias ! xs:integer(.))
    
    (:Filtramos para quedarnos solo con pacientes que tengan historial O bajas previas :)
    where $num_eventos > 0 or $total_dias_baja > 0
    
    (: Ordenamos por gravedad del triaje (los más graves primero) :)
    order by $nivel ascending
    
    (:Generamos la estructura XML de salida  :)
    return
    <Paciente hash_id="{$id}" prioridad_triaje="{$nivel}">
        <Episodio_Actual>
            <Motivo>{$motivo}</Motivo>
        </Episodio_Actual>
        <Antecedentes>
            <Eventos_Respiratorios_Previos>{$num_eventos}</Eventos_Respiratorios_Previos>
            <Dias_Acumulados_Baja>{$total_dias_baja}</Dias_Acumulados_Baja>
        </Antecedentes>
    </Paciente>
}
</Informe_Pacientes>




xquery version "3.1";

let $historiales := doc("/db/PROYECTO/historiales_semanticos_maximo.xml")

return
<Pacientes_Alto_Riesgo>
{
    for $paciente in $historiales//Paciente
    
    let $id := $paciente/@id
    let $num_eventos := count($paciente/Fichero_Eventos/Evento_Medico)
    
    (: Filtramos a los que tienen 3 o más episodios respiratorios :)
    where $num_eventos >= 3
    
    order by $num_eventos descending
    
    return
    <Paciente_Cronico>
        <Hash_ID>{$id}</Hash_ID>
        <Total_Episodios_Registrados>{$num_eventos}</Total_Episodios_Registrados>
        <Nivel_Riesgo>ALTO</Nivel_Riesgo>
    </Paciente_Cronico>
}
</Pacientes_Alto_Riesgo>





xquery version "3.1";

let $urgencias := doc("/db/PROYECTO/urgencias_criticas_semanticas.xml")

return
<Monitor_Saturacion_Geriatrica>
{
    for $urg in $urgencias//Paciente_Critico
    
    let $id := $urg/@identificador
    let $sat_str := $urg/Estado_Clinico_Actual/Saturacion_O2/text()
    let $anio_nac := $urg/Anio_Nacimiento/text()
    
    (: Convertimos la saturación a decimal :)
    let $sat_num := if ($sat_str castable as xs:double) then xs:double($sat_str) else 100.0
    
    (: Filtramos a los que tienen menos del 95% de oxígeno :)
    where $sat_num < 95.0
    
    order by $sat_num ascending
    
    return
    <Alerta_Hipoxia>
        <Paciente_Hash>{substring($id, 1, 15)}...</Paciente_Hash>
        <Edad_Estimada>{2026 - xs:integer($anio_nac)} años</Edad_Estimada>
        <Nivel_Oxigeno>{$sat_num} %</Nivel_Oxigeno>
        <Estado_Triaje>{$urg/Estado_Clinico_Actual/Motivo_Principal/text()}</Estado_Triaje>
    </Alerta_Hipoxia>
}
</Monitor_Saturacion_Geriatrica>
