SELECT 
    pa.codigo_hash,
    pa.anio_nacimiento,
    pa.sexo_biologico,
    pa.grupo_sanguineo,
    pa.region_sanitaria,
    ec.fecha_evento,
    ec.tipo_evento,
    ec.codigo_cie10,
    ec.descripcion_cie10,
    ec.especialidad
FROM paciente_anonimo pa
INNER JOIN evento_clinico ec ON ec.id_paciente = pa.id_paciente
WHERE ec.codigo_cie10 LIKE 'J%' -- Filtramos por respiratorio para acotar el volumen
ORDER BY pa.codigo_hash, ec.fecha_evento ASC;


SELECT 
    codigo_hash_paciente,
    fecha_inicio_baja,
    dias_totales_baja,
    codigo_cie10_causa,
    tipo_contingencia
FROM historico_bajas_medicas
ORDER BY codigo_hash_paciente;


USE PROYECTO;

SELECT 
    pa.codigo_hash,
    pa.anio_nacimiento,
    pa.sexo_biologico,
    t.fecha_hora_llegada,
    t.nivel_triaje_manchester,
    t.motivo_consulta,
    t.saturacion_oxigeno,
    t.estado_actual
FROM paciente_anonimo pa
INNER JOIN triaje_urgencias_actual t ON pa.codigo_hash = t.codigo_hash_paciente
WHERE pa.anio_nacimiento <= 1966 
  AND t.nivel_triaje_manchester <= 2
ORDER BY t.nivel_triaje_manchester ASC, t.fecha_hora_llegada ASC;




xquery version "3.1";

let $historiales := doc("/db/PROYECTO/historiales_semanticos_maximo.xml")
let $bajas := doc("/db/PROYECTO/bajas_semanticas.xml")

return
<Impacto_Laboral_Respiratorio>
{
    (: Recorremos todos los pacientes del historial clínico :)
    for $paciente in $historiales//Paciente
    
    let $id := $paciente/@id
    let $num_eventos := count($paciente/Fichero_Eventos/Evento_Medico)
    
    (: Buscamos las bajas de este paciente específico :)
    let $bajas_paciente := $bajas//Expediente_Laboral[@id_paciente = $id]/Incapacidad_Temporal
    let $total_dias := sum($bajas_paciente/Duracion_Dias ! xs:integer(.))
    
    (: Filtramos Solo crónicos (>=2 eventos) que además hayan causado baja :)
    where $num_eventos >= 2 and $total_dias > 0
    
    (: Ordenamos por los que más días de baja acumulan :)
    order by $total_dias descending
    
    return
    <Paciente_Cronico id_hash="{$id}">
        <Total_Episodios>{$num_eventos}</Total_Episodios>
        <Dias_Totales_Baja>{$total_dias}</Dias_Totales_Baja>
        <Media_Dias_Por_Episodio>{round($total_dias div $num_eventos)}</Media_Dias_Por_Episodio>
    </Paciente_Cronico>
}
</Impacto_Laboral_Respiratorio>
